

#include <iostream>
#include <cmath> //Подключение библиотек Math

using namespace std;

int main()
{
    double R1, R2, S1, S2;
    
    cout << "Введите два вещественных числа" << endl; //Выводим сообщение. << endl - перенос на новую строку
    cin >> R1 >> R2; //просим ввести два числа по очереди
    //Равносильно записи cin >> R1; cin >> R2;
    
    S1 = M_PI * pow(R1, 2); //Высчитываем прощадь первого радиуса. Pow - вводит в степень 2, M_PI - константа числа ПИ
    S2 = M_PI * pow(R2, 2); //Высчитываем площадь второго радиуса.
    
    if(S1 < S2) { //Проверка больше ли первый круг, чем второй
        cout << "Второе число больше первого" << endl; //Если нет, выводим сообщение
        return 0; // выходим из этого блока
    }
    
    cout << "Площадь кольца = " << S1 - S2 << endl; //Если мы ещё не выкинули программу, то выводим S1 - S2

    return 0; // тут выходим.
}
