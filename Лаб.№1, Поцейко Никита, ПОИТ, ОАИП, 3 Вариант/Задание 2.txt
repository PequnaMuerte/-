

#include <iostream>
#include <cmath> //Подключение библиотек Math
using namespace std;

int main()
{
    int number = 135, summ = 0;
    summ += number % 10; //Суммируем сумму с остатком деления числа на 10
    //Равносильно записи summ = summ + number % 10;
    cout << "Количество единиц в числе: " << number % 10 << endl; //выводи остаток от деления
    number /= 10; //делим число на 10. Равносильно number = number = 10. Теперь число не 135, а 13.
    summ += number % 10; //Делаем тоже самое, что сначала
    cout << "Количество десятков в числе: " << number % 10 << endl; //Выводим остаток от деления, но уже от 13
    number /= 10; //опять делим. В итоге у нас остаются лишь сотни.
    summ += number; //складываем
    cout << "Количество сотен в числе: " << number << endl; //Выводим количество сотен
    cout << "Сумма цифр числа: " << summ; //Выводим сумму цифр числа
    return 0; //тут выходим 
}
